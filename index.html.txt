<body>
    <h1>Weekly Timesheet</h1>

    <div class="week-nav">
        <button onclick="changeWeek(-7)">Previous Week</button>
        <button onclick="changeWeek(7)">Next Week</button>
    </div>

    <div id="daysContainer"></div>

    <div id="totalHours">Total: 0 hours</div>

    <button class="email-btn" onclick="sendEmail()">Email Timesheet</button>

    <script>
        const defaultStart = "08:00";
        const defaultFinish = "16:00";

        const days = [
            "Monday","Tuesday","Wednesday","Thursday","Friday","Saturday","Sunday"
        ];

        function getMonday(date) {
            const d = new Date(date);
            const day = d.getDay();
            const diff = (day === 0 ? -6 : 1) - day;
            d.setDate(d.getDate() + diff);
            return d;
        }

        let currentWeekStart = getMonday(new Date());

        function changeWeek(offsetDays) {
            currentWeekStart.setDate(currentWeekStart.getDate() + offsetDays);
            renderWeek();
        }

        function renderWeek() {
            const container = document.getElementById("daysContainer");
            container.innerHTML = "";

            days.forEach((day, i) => {
                const date = new Date(currentWeekStart);
                date.setDate(date.getDate() + i);

                const label = date.toLocaleDateString("en-GB", {
                    weekday: "long",
                    day: "2-digit",
                    month: "short"
                });

                const div = document.createElement("div");
                div.className = "day";

                div.innerHTML = `
                    <h2>${label}</h2>
                    <label>Start Time:</label>
                    <input type="time" id="start-${i}" value="${defaultStart}" onchange="updateTotal()" />
                    <label>Finish Time:</label>
                    <input type="time" id="finish-${i}" value="${defaultFinish}" onchange="updateTotal()" />
                `;

                container.appendChild(div);
            });

            updateTotal();
        }

        function calculateHours(start, finish) {
            if (!start || !finish) return 0;
            const [sh, sm] = start.split(":").map(Number);
            const [fh, fm] = finish.split(":").map(Number);
            return (fh + fm/60) - (sh + sm/60);
        }

        function updateTotal() {
            let total = 0;

            days.forEach((day, i) => {
                const s = document.getElementById(`start-${i}`).value;
                const f = document.getElementById(`finish-${i}`).value;
                total += calculateHours(s, f);
            });

            document.getElementById("totalHours").textContent =
                `Total: ${total.toFixed(2)} hours`;
        }

        function sendEmail() {
            let body = `Timesheet for week starting ${currentWeekStart.toLocaleDateString("en-GB")}%0D%0A%0D%0A`;

            days.forEach((day, i) => {
                const date = new Date(currentWeekStart);
                date.setDate(date.getDate() + i);

                const label = date.toLocaleDateString("en-GB", {
                    weekday: "long",
                    day: "2-digit",
                    month: "short"
                });

                const s = document.getElementById(`start-${i}`).value;
                const f = document.getElementById(`finish-${i}`).value;
                const hrs = calculateHours(s, f).toFixed(2);

                body += `${label}: ${s} - ${f} (${hrs} hrs)%0D%0A`;
            });

            const total = document.getElementById("totalHours").textContent;
            body += `%0D%0A${total}`;

            window.location.href = `mailto:?subject=Weekly Timesheet&body=${body}`;
        }

        renderWeek();
    </script>
